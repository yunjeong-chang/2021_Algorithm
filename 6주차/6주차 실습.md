**1번. 이진탐색 - 재귀**
```c
# include <stdio.h>
# include <stdlib.h>
# define swap(x, y) do {int tmp = x; x = y; y = tmp;}while(0)
# pragma warning(disable:4996)

int BinarySearch(int* arr, int start, int end, int key) {
	if (start > end) {
		return end;
	}
	int mid = (start + end) / 2;

	if (arr[mid] < key) {
		BinarySearch(arr, mid + 1, end, key);
	}
	else if (arr[mid] > key) {
		BinarySearch(arr, start, mid - 1, key);
	}
	else
		return mid;
}

int main() {
	int n, k;
	scanf("%d %d", &n, &k); //배열의 크기 n, 탐색할 키 k

	int* arr = (int*)malloc(sizeof(int) * n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &arr[i]);
	}
	printf(" %d", BinarySearch(arr, 0, n - 1, k));
	return 0;
}
```

**2번. 이진탐색 - 비재귀**
```c
# include <stdio.h>
# include <stdlib.h>
# define swap(x, y) do {int tmp = x; x = y; y = tmp;}while(0)
# pragma warning(disable:4996)

int BinarySearch2(int* arr, int start, int end, int key) { //이진탐색 비재귀
	int mid = (start + end) / 2;

	while (start <= end) {
		if (arr[mid] < key) {
			start = mid + 1;
			mid = (mid + 1 + end) / 2;
		}
		else if (arr[mid] > key) {
			end = mid - 1;
			mid = (start + mid - 1) / 2;
		}
		else
			return mid;
	}

	return start; //end로 바꾸면 문제 1의 상황
}


int main() {
	int n, k;
	scanf("%d %d", &n, &k); //배열의 크기 n, 탐색할 키 k

	int* arr = (int*)malloc(sizeof(int) * n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &arr[i]);
	}
	printf(" %d", BinarySearch2(arr, 0, n - 1, k));
	return 0;
}
```

**3번. 이진탐색응용 - 게임**
```c
# include <stdio.h>
# include <stdlib.h>
# define swap(x, y) do {int tmp = x; x = y; y = tmp;}while(0)
# pragma warning(disable:4996)

int BinarySearchSejong(int a, int b, int n, char* str) {
	for (int i = 0; i < n; i++) {
		int mid = (a + b) / 2;
		if (str[i] == 'Y') {
			a = mid + 1;
		}
		else if (str[i] == 'N') {
			b = mid;
		}
	}
	return a;
}

int main() {
	int a, b, n;
	scanf("%d %d %d", &a, &b, &n);
	getchar();

	char str[100];
	scanf("%s", str);

	printf("%d", BinarySearchSejong(a, b, n, str));
	return 0;
}
```
