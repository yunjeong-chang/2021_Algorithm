**1번. 삽입식 힙 **
```c
# include <stdio.h>
# include <stdlib.h>
# define swap(x, y) do {int tmp = x; x = y; y = tmp;}while(0)
# pragma warning(disable:4996)

int H[100]; //힙
int n = 0;  //힙의 크기

void upHeap(int i) {
	while (i > 1 && H[i] > H[i / 2]) {
		swap(H[i], H[i / 2]);
		i /= 2;
	}
	return;
}

void downHeap(int i) {
	int index = i, child;
	while (index <= n / 2) {
		child = index * 2;
		if (child + 1 <= n && H[child] < H[child + 1]) {
			child++;
		}
		if (H[child] > H[index]) {
			swap(H[child], H[index]);
			index = child;
		}
		else break;
	}
	return;
}

void insertItem(int key) {
	n++;
	H[n] = key; 
	upHeap(n); 
	return;
}

int removeMax() {
	int max = H[1]; 
	H[1] = H[n]; 
	H[n] = 0; 
	n--; 
	downHeap(1); 
	return max;
}

void printHeap() {
	for (int i = 1; i <= n; i++)
		printf(" %d", H[i]);
	printf("\n");
	return;
}

int main() {
	char order;
	int key;

	while (1) {
		scanf("%c", &order);
		getchar();
		
		if (order == 'q') break;

		else if (order == 'i') {
			scanf("%d", &key);
			getchar();
			insertItem(key);
			printf("0\n");
		}

		else if (order == 'd') {
			printf("%d\n", removeMax());
		}

		else if (order == 'p') {
			printHeap();
		}
	}
	return 0;
}
```

**2번. 상향식 **
```c
# include <stdio.h>
# include <stdlib.h>
# define swap(x, y) do {int tmp = x; x = y; y = tmp;}while(0)
# pragma warning(disable:4996)

int H[100]; //힙
int n = 0;  //힙의 크기

void downHeap(int i) {
	int index = i, child;
	while (index <= n / 2) {
		child = index * 2;
		if (child + 1 <= n && H[child] < H[child + 1]) {
			child++;
		}
		if (H[child] > H[index]) {
			swap(H[child], H[index]);
			index = child;
		}
		else break;
	}
	return;
}

void rBuildHeap(int i) {
	if (i > n) return;

	rBuildHeap(2 * i);
	rBuildHeap(2 * i + 1);
	downHeap(i);

	return;
}

void buildHeap() {

	for (int i = n / 2; i > 0; i--) {
		downHeap(i);
	}

	return;
}

void printHeap() {
	for (int i = 1; i <= n; i++) {
		printf(" %d", H[i]);
	}
	printf("\n");
	return;
}

int main() {

	scanf("%d", &n);

	for (int i = 1; i <= n; i++) {
		scanf("%d", &H[i]);
	}

	if (n % 2 == 0)
		rBuildHeap(1); 
	else
		buildHeap();   

	printHeap();
	return 0;
}
```
